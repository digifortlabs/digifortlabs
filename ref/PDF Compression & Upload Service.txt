const { S3Client, PutObjectCommand } = require("@aws-sdk/client-s3");
const zlib = require("zlib");
const { promisify } = require("util");
const gzip = promisify(zlib.gzip);

// Configure the S3 Client for the Mumbai region
const s3Client = new S3Client({ region: "ap-south-1" });

/**
 * Compresses a PDF buffer and uploads it to AWS S3.
 * Then logs the metadata to the PostgreSQL database.
 * * @param {Buffer} fileBuffer - The original PDF file buffer
 * @param {Object} metadata - Contextual data (hospitalId, patientId, userId, uhid)
 */
async function compressAndUploadMRD(fileBuffer, metadata) {
  const { hospitalId, patientId, userId, uhid, docType } = metadata;
  const originalSize = fileBuffer.length;

  try {
    // 1. Perform Compression (Gzip)
    // Note: For medical PDFs, Gzip provides safe lossless compression for storage
    console.log(`Starting compression for Patient: ${uhid}...`);
    const compressedBuffer = await gzip(fileBuffer);
    const compressedSize = compressedBuffer.length;
    const savings = ((originalSize - compressedSize) / originalSize * 100).toFixed(2);

    // 2. Define S3 Key (Path)
    // Structure: hospitals/{hospitalId}/{patientId}/{timestamp}_{filename}.pdf.gz
    const timestamp = Date.now();
    const s3Key = `hospitals/${hospitalId}/${patientId}/${timestamp}_record.pdf.gz`;

    // 3. Upload to AWS S3
    const uploadParams = {
      Bucket: "digifort-labs-mrd-india",
      Key: s3Key,
      Body: compressedBuffer,
      ContentType: "application/x-gzip", // Indicate it's compressed
      ContentEncoding: "gzip",
      Metadata: {
        "original-size": originalSize.toString(),
        "patient-uhid": uhid,
      },
      ServerSideEncryption: "AES256", // Enforce encryption per policy
    };

    await s3Client.send(new PutObjectCommand(uploadParams));
    console.log(`Upload successful. S3 Key: ${s3Key}`);

    // 4. Save Metadata to Database (Matches database_schema.sql)
    const dbQuery = `
      INSERT INTO medical_records (
        patient_id, 
        hospital_id, 
        document_type, 
        s3_key, 
        is_compressed, 
        original_size_kb, 
        compressed_size_kb, 
        compression_method, 
        uploaded_by, 
        visit_date
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, CURRENT_DATE)
      RETURNING id;
    `;

    const values = [
      patientId,
      hospitalId,
      docType,
      s3Key,
      true,
      Math.round(originalSize / 1024),
      Math.round(compressedSize / 1024),
      'gzip',
      userId
    ];

    // Example DB execution (assuming a 'pg' client is initialized)
    // const res = await db.query(dbQuery, values);
    
    return {
      success: true,
      s3Key,
      savings: `${savings}%`,
      originalSizeKb: Math.round(originalSize / 1024),
      compressedSizeKb: Math.round(compressedSize / 1024)
    };

  } catch (error) {
    console.error("Critical Error in MRD Upload Pipeline:", error);
    throw error;
  }
}

module.exports = { compressAndUploadMRD };
