import React, { useState, useRef } from 'react';
import { 
  Upload, 
  FileText, 
  CheckCircle2, 
  AlertCircle, 
  Database, 
  ArrowRight,
  RefreshCw,
  Info
} from 'lucide-react';

const App = () => {
  const [file, setFile] = useState(null);
  const [status, setStatus] = useState('idle'); // idle, uploading, success, error
  const [progress, setProgress] = useState(0);
  const [results, setResults] = useState(null);
  const fileInputRef = useRef(null);

  const handleFileChange = (e) => {
    const selectedFile = e.target.files[0];
    if (selectedFile && selectedFile.type === 'application/pdf') {
      setFile(selectedFile);
      setStatus('idle');
      setResults(null);
    } else {
      // Use a custom message UI instead of alert()
      setStatus('error');
    }
  };

  const simulateUpload = () => {
    if (!file) return;
    
    setStatus('uploading');
    setProgress(0);

    // Simulate upload and compression progress
    const interval = setInterval(() => {
      setProgress((prev) => {
        if (prev >= 100) {
          clearInterval(interval);
          completeProcess();
          return 100;
        }
        return prev + 5;
      });
    }, 100);
  };

  const completeProcess = () => {
    // Mocking the compression result based on the Node.js logic
    const originalSizeKb = Math.round(file.size / 1024);
    const compressedSizeKb = Math.round(originalSizeKb * 0.45); // Simulate 55% savings
    const savings = (((originalSizeKb - compressedSizeKb) / originalSizeKb) * 100).toFixed(1);

    setResults({
      original: originalSizeKb,
      compressed: compressedSizeKb,
      savings: savings,
      s3Path: `hospitals/H001/${file.name.replace(/\s/g, '_')}.pdf.gz`
    });
    setStatus('success');
  };

  const reset = () => {
    setFile(null);
    setStatus('idle');
    setProgress(0);
    setResults(null);
  };

  return (
    <div className="min-h-screen bg-slate-50 flex items-center justify-center p-6 font-sans">
      <div className="max-w-xl w-full bg-white rounded-[2.5rem] shadow-2xl border border-slate-100 overflow-hidden">
        {/* Header */}
        <div className="bg-indigo-950 p-8 text-white relative">
          <div className="relative z-10">
            <h1 className="text-2xl font-black mb-1">Digital MRD Upload</h1>
            <p className="text-indigo-300 text-sm">Securely compress and archive patient records to AWS Mumbai</p>
          </div>
          <Database className="absolute -bottom-4 -right-4 w-32 h-32 text-indigo-800 opacity-20 transform rotate-12" />
        </div>

        <div className="p-8">
          {status === 'idle' && (
            <div 
              onClick={() => fileInputRef.current.click()}
              className="border-2 border-dashed border-slate-200 rounded-3xl p-12 text-center hover:border-indigo-500 hover:bg-indigo-50/30 transition-all cursor-pointer group"
            >
              <input 
                type="file" 
                hidden 
                ref={fileInputRef} 
                onChange={handleFileChange} 
                accept=".pdf"
              />
              <div className="bg-slate-100 w-16 h-16 rounded-2xl flex items-center justify-center mx-auto mb-4 group-hover:scale-110 transition-transform">
                <Upload className="text-slate-400 group-hover:text-indigo-600" size={28} />
              </div>
              <h3 className="font-bold text-slate-800 mb-1">
                {file ? file.name : "Choose Patient PDF"}
              </h3>
              <p className="text-slate-400 text-xs uppercase tracking-widest font-semibold">
                Max size: 50MB • HIPAA Compliant
              </p>
            </div>
          )}

          {status === 'uploading' && (
            <div className="space-y-6 py-8 text-center">
              <div className="w-20 h-20 bg-indigo-100 rounded-full flex items-center justify-center mx-auto relative overflow-hidden">
                <div 
                  className="absolute bottom-0 left-0 w-full bg-indigo-500 transition-all duration-300 ease-out opacity-20" 
                  style={{ height: `${progress}%` }}
                />
                <RefreshCw className="text-indigo-600 animate-spin" size={32} />
              </div>
              <div>
                <h3 className="text-xl font-black text-slate-800 mb-1">Processing...</h3>
                <p className="text-slate-500 text-sm">Compressing and encrypting clinical data</p>
              </div>
              <div className="space-y-2">
                <div className="w-full bg-slate-100 h-2.5 rounded-full overflow-hidden">
                  <div 
                    className="bg-indigo-600 h-full rounded-full transition-all duration-300 shadow-[0_0_10px_rgba(79,70,229,0.4)]" 
                    style={{ width: `${progress}%` }}
                  />
                </div>
                <p className="text-right text-[10px] font-bold text-slate-400 uppercase tracking-widest">
                  {progress}% Complete
                </p>
              </div>
            </div>
          )}

          {status === 'success' && results && (
            <div className="animate-in fade-in slide-in-from-bottom-4 duration-500">
              <div className="flex items-center gap-4 mb-8 bg-emerald-50 p-4 rounded-2xl border border-emerald-100">
                <div className="bg-emerald-500 p-2 rounded-xl">
                  <CheckCircle2 size={24} className="text-white" />
                </div>
                <div>
                  <h3 className="font-bold text-emerald-900">Upload Finalized</h3>
                  <p className="text-emerald-700 text-xs">Record indexed and stored securely.</p>
                </div>
              </div>

              <div className="grid grid-cols-2 gap-4 mb-8">
                <div className="bg-slate-50 p-5 rounded-2xl border border-slate-100">
                  <p className="text-slate-400 text-[10px] uppercase tracking-widest font-bold mb-1">Original Size</p>
                  <p className="text-xl font-black text-slate-800">{results.original} <span className="text-sm font-medium">KB</span></p>
                </div>
                <div className="bg-indigo-50 p-5 rounded-2xl border border-indigo-100">
                  <p className="text-indigo-400 text-[10px] uppercase tracking-widest font-bold mb-1">Compressed Size</p>
                  <p className="text-xl font-black text-indigo-700">{results.compressed} <span className="text-sm font-medium">KB</span></p>
                </div>
              </div>

              <div className="bg-indigo-600 rounded-3xl p-6 text-white mb-8 relative overflow-hidden">
                <div className="relative z-10 flex justify-between items-center">
                  <div>
                    <h4 className="text-3xl font-black">{results.savings}%</h4>
                    <p className="text-indigo-200 text-xs font-medium">Storage Space Saved</p>
                  </div>
                  <div className="bg-white/10 p-3 rounded-2xl backdrop-blur-md">
                    <FileText size={32} />
                  </div>
                </div>
                <div className="absolute -left-4 -top-4 w-24 h-24 bg-white/5 rounded-full blur-2xl"></div>
              </div>

              <div className="space-y-2 mb-8 p-4 bg-slate-50 rounded-2xl border border-slate-100">
                <p className="text-[10px] font-bold text-slate-400 uppercase flex items-center gap-1.5">
                  <Info size={12} /> S3 Storage Identity
                </p>
                <code className="text-[10px] text-slate-500 block break-all font-mono leading-relaxed">
                  {results.s3Path}
                </code>
              </div>
            </div>
          )}

          {status === 'error' && (
            <div className="text-center py-8">
              <div className="bg-red-50 w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4 border border-red-100">
                <AlertCircle className="text-red-500" size={28} />
              </div>
              <h3 className="font-bold text-slate-800">Invalid File Type</h3>
              <p className="text-slate-500 text-sm mb-6">Digifort Labs strictly supports PDF medical records for archiving.</p>
              <button 
                onClick={() => setStatus('idle')}
                className="text-indigo-600 font-bold text-sm hover:underline"
              >
                Try again
              </button>
            </div>
          )}

          {/* Actions */}
          <div className="flex gap-3 mt-4">
            {status === 'idle' && (
              <button 
                disabled={!file}
                onClick={simulateUpload}
                className={`w-full py-4 rounded-2xl font-black text-sm transition-all flex items-center justify-center gap-2 ${
                  file 
                  ? 'bg-indigo-600 text-white shadow-xl shadow-indigo-100 hover:bg-indigo-700 active:scale-[0.98]' 
                  : 'bg-slate-100 text-slate-400 cursor-not-allowed'
                }`}
              >
                Start Archiving <ArrowRight size={18} />
              </button>
            )}
            {status === 'success' && (
              <button 
                onClick={reset}
                className="w-full py-4 rounded-2xl bg-slate-100 text-slate-700 font-black text-sm hover:bg-slate-200 transition-all active:scale-[0.98]"
              >
                Upload Another Record
              </button>
            )}
          </div>
        </div>
      </div>
    </div>
  );
};

export default App;
