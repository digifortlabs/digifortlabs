-- Digifort Labs: Multi-Tenant MRD System Schema
-- Designed for PostgreSQL (AWS RDS)

-- 1. Hospitals Table (Tenants)
CREATE TABLE hospitals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    branch_code VARCHAR(50) UNIQUE NOT NULL, -- e.g., MUM-01, GUJ-04
    location VARCHAR(255),
    registration_no VARCHAR(100), -- Hospital Govt Reg No
    subscription_plan VARCHAR(50) DEFAULT 'Starter', -- Starter, Pro, Enterprise
    status VARCHAR(20) DEFAULT 'Active', -- Active, Suspended, Pending
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Users Table (RBAC)
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    hospital_id UUID REFERENCES hospitals(id) ON DELETE CASCADE,
    full_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role VARCHAR(50) NOT NULL, -- SuperAdmin, HospitalAdmin, MRDClerk, Doctor
    last_login TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 3. Patients Table
CREATE TABLE patients (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    hospital_id UUID REFERENCES hospitals(id) ON DELETE CASCADE,
    uhid VARCHAR(100) NOT NULL, -- Universal Hospital ID
    abha_id VARCHAR(14), -- Ayushman Bharat Health Account ID
    patient_name VARCHAR(255) NOT NULL,
    date_of_birth DATE,
    gender VARCHAR(20),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(hospital_id, uhid) -- Ensures UHID is unique per hospital
);

-- 4. Digital Records (PDF Management)
CREATE TABLE medical_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
    hospital_id UUID REFERENCES hospitals(id) ON DELETE CASCADE,
    document_type VARCHAR(100), -- Discharge Summary, Lab Report, Scan
    s3_key TEXT NOT NULL, -- Path in AWS S3
    uploaded_by UUID REFERENCES users(id),
    visit_date DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 5. Physical File Tracking
CREATE TABLE physical_files (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID REFERENCES patients(id) ON DELETE CASCADE,
    hospital_id UUID REFERENCES hospitals(id) ON DELETE CASCADE,
    aisle_no VARCHAR(50),
    rack_no VARCHAR(50),
    shelf_no VARCHAR(50),
    box_no VARCHAR(50),
    current_status VARCHAR(50) DEFAULT 'In-MRD', -- In-MRD, With-Doctor, Lost
    last_moved_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 6. Subscription Billing
CREATE TABLE subscriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    hospital_id UUID REFERENCES hospitals(id) ON DELETE CASCADE,
    amount_inr DECIMAL(10, 2),
    billing_cycle VARCHAR(20), -- Monthly, Annual
    next_billing_date DATE,
    payment_status VARCHAR(20) DEFAULT 'Paid'
);

-- INDEXES for Performance (Crucial for Multi-Tenancy)
CREATE INDEX idx_hospital_patients ON patients(hospital_id);
CREATE INDEX idx_hospital_records ON medical_records(hospital_id);
CREATE INDEX idx_patient_uhid ON patients(uhid);
